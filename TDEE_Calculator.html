    <script>

        function calculateTDEE() {

            const weight = parseFloat(document.getElementById('weight').value);

            const height = parseFloat(document.getElementById('height').value);

            const age = parseInt(document.getElementById('age').value);

            const sex = document.getElementById('sex').value;

            const activity = parseFloat(document.getElementById('activity').value);


            if (!weight || !height || !age) {

                document.getElementById('results').innerHTML = '<p>Please fill out all fields correctly.</p>';

                return;

            }


            let BMR;

            if (sex === 'male') {

                BMR = 66 + (6.23 * weight) + (12.7 * height) - (6.8 * age);

            } else {

                BMR = 655 + (4.35 * weight) + (4.7 * height) - (4.7 * age);

            }


            const TDEE = BMR * activity;

            const deficitCalories = TDEE - 500; // Typical cutting deficit


            document.getElementById('results').innerHTML = `

                <h3>Results:</h3>

                <p><strong>BMR:</strong> ${BMR.toFixed(2)} calories/day</p>

                <p><strong>TDEE:</strong> ${TDEE.toFixed(2)} calories/day</p>

                <p><strong>Cutting Calories:</strong> ${deficitCalories.toFixed(2)} calories/day</p>

                <h4>Macronutrient Recommendations (for cutting):</h4>

                <ul>

                    <li><strong>Protein:</strong> ${(0.8 * weight).toFixed(2)} - ${(1.0 * weight).toFixed(2)} grams/day</li>

                    <li><strong>Fats:</strong> ${(0.25 * deficitCalories / 9).toFixed(2)} grams/day</li>

                    <li><strong>Carbohydrates:</strong> ${(deficitCalories - (0.8 * weight * 4) - (0.25 * deficitCalories)).toFixed(2) / 4} grams/day</li>

                </ul>

            `;

        }

    </script>
